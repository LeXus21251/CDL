
PB_Fire_Super_Event = {
	hidden_effect = {
		if = {
			limit = { is_ai = no }
			set_variable = { Active_Superevent_ID = super_id }
				meta_effect = {
					text = {
						scoped_play_song = "Super_Event_Sound_[ID]"
					}
					ID = "[?super_id]"
				}
	}
}

Console_Fire_Super_Event = {
	set_temp_variable = { super_id = args^0 }
	PB_Fire_Super_Event = yes
}
