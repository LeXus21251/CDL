scripted_gui = {
    CDL_politics_buttons = {
        window_name = "politics_buttons_window"
        context_type = player_context
		parent_window_token = politics_tab
         
        visible = {
            has_global_flag = politics_buttons_visible
        }
        ## SECOND LEADER ##
        effects = {
			leaders_button_click = {
				if = {
					limit = {
                        has_variable = second_leader_pol_open
                    }
					clear_variable = second_leader_pol_open
				}
				else = {
					set_variable = { second_leader_pol_open = 1 }
				}
			}
        }

		triggers = {
			leaders_button_click_enabled = {
				has_country_flag = has_second_leader
			}
		}

		## POLITICAL PARTIES ##



		effects = {
			political_parties_button_click = {
				if = {
					limit = {
                        has_variable = pol_parties_open
                    }
					clear_variable = pol_parties_open
				}
				else = {
					set_variable = { pol_parties_open = 1 }
				}
			}
        }

		triggers = {
			political_parties_button_click_enabled = {
				always = yes
			}
		}
}
}
