on_actions = {
	on_startup = { 
		effect = {
			ENG = { # Don't mind this, scope fix.
				CDL_cultures = yes
			}
			ROOT = {
				set_variable = { PB_Active_Conversions = 0 }
			}
		} 
	}
	on_state_control_changed = { 
		effect = {
			ROOT = {
				every_owned_state = {
					limit = {
						NOT = {
							is_in_array = {
								array = ROOT.CDL_accepted_cultures
								value = culture_id
							}
							has_dynamic_modifier = {
								modifier = Foreign_Culture
							}
						}	
					}
					add_dynamic_modifier = {
						modifier = Foreign_Culture
					}
				}
				every_owned_state = {
					limit = {
						is_in_array = {
							array = ROOT.CDL_accepted_cultures
							value = culture_id
						}
						has_dynamic_modifier = {
							modifier = Foreign_Culture
						}	
					}
					remove_dynamic_modifier	 = {
						modifier = Foreign_Culture
					}
				}
				every_owned_state = {
					limit = {
						is_in_array = {
							array = ROOT.CDL_accepted_cultures
							value = culture_id
						}
						has_dynamic_modifier = {
							modifier = Culture_Conversion
						}	
					}
					remove_dynamic_modifier	 = {
						modifier = Culture_Conversion
					}
				}
			}
		} 
	}
	on_startup = { 
		effect = {
			ROOT = {
				every_state = {
					limit = {
						NOT = {
   							is_in_array = {
        						array = ROOT.CDL_accepted_cultures
        						value = culture_id
    						}
						}
					}
					add_dynamic_modifier = {
   						modifier = Foreign_Culture
					}
				}
			}
		} 
	}
}
